class Remote
{
    protected $http;
    protected $url;

    public function __construct(Guzzle $http, string $url)
    {
        $this->http = $http;
        $this->url = $url;
    }

    public function __sleep()
    {
        return array_keys(
            array_except(
                get_object_vars($this),
                'http'
            )
        );
    }

    public function __wakeup()
    {
        $this->http = new Guzzle();
    }
}
